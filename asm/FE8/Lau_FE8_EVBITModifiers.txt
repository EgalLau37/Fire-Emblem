/***\
EVBIT ID 0x7
Related to cutscene fading; if toggled on, the game automatically
fades to dark after an event finishes.
Default value is on (fade after event).
This can be toggled to either invert or negate the flag.
\***/


#ifdef EVBIT7_invert
ORG 0x592150
BYTE 0x40
#endif

#ifdef EVBIT7_default
ORG 0x592150
BYTE 0x41
#endif

#ifdef EVBIT7_always
ORG 0x592150
BYTE 0x45
#endif

#ifdef EVBIT7_never
ORG 0x592150
BYTE 0x40
ORG 0x592154
BYTE 0xD
#endif


//---------------------------------------------------------------//

/***\
EVBIT ID 0x9
Related to camera movement; if toggled on, the camera follows
units as they move; the opposite behavior of default FE7.
Default value is off (do not follow units).
This can be toggled to either invert or negate the flag.
\***/

#ifdef EVBIT9_invert
ORG 0xFCF0
SHORT 0x01D1
ORG 0xFCA2
SHORT 0x0CD1
#endif

#ifdef EVBIT9_default
ORG 0xFCF0
SHORT 0x01D0
ORG 0xFCA2
SHORT 0x0CD0
#endif

#ifdef EVBIT9_always
ORG 0xFCF0
SHORT 0x1B1C
ORG 0xFCA2
SHORT 0x1B1C
#endif

#ifdef EVBIT9_never
ORG 0xFCF0
SHORT 0x01E0
ORG 0xFCA2
SHORT 0x0CE0
#endif
